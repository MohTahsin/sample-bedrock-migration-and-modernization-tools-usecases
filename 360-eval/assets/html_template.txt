      const selectedChart = document.getElementById('radar-' + taskName);

        selectedChart.style.display = 'block';
        // Force Plotly to resize
        window.dispatchEvent(new Event('resize'));
      }

    function showTaskTable(taskName) {
      console.log('Switching to table:', taskName); // Debug logging
      
      // Hide all task tables
      document.querySelectorAll('.task-table').forEach(function (el) {
        el.style.display = 'none';
      });
      
      // Show selected task's table - match the ID pattern used in HTML template
      const selectedTable = document.getElementById('table-' + taskName);
      console.log('Found table element:', selectedTable); // Debug logging
      
      if (selectedTable) {
        selectedTable.style.display = 'block';
        
        // Force Plotly tables to re-render properly
        setTimeout(function() {
          // Try multiple approaches to re-render Plotly content
          const plotlyElements = selectedTable.querySelectorAll('div[class*="plotly"]');
          console.log('Found Plotly elements:', plotlyElements.length); // Debug logging
          
          plotlyElements.forEach(function(element) {
            // Method 1: Standard Plotly resize
            if (window.Plotly && element._fullLayout) {
              try {
                window.Plotly.Plots.resize(element);
              } catch (e) {
                console.log('Plotly resize failed:', e);
              }
            }
            
            // Method 2: Force redraw by manipulating the element
            if (window.Plotly && element._fullData) {
              try {
                window.Plotly.redraw(element);
              } catch (e) {
                console.log('Plotly redraw failed:', e);
              }
            }
          });
          
          // Method 3: Force reflow by temporarily changing display
          selectedTable.style.display = 'none';
          setTimeout(function() {
            selectedTable.style.display = 'block';
            
            // Final resize event
            window.dispatchEvent(new Event('resize'));
            
            // Method 4: Force text repaint by touching font styles
            const textElements = selectedTable.querySelectorAll('text, span, div');
            textElements.forEach(function(el) {
              if (el.style) {
                const originalColor = el.style.color || '';
                el.style.color = '#333333';
                setTimeout(function() {
                  el.style.color = originalColor || '#333333';
                }, 1);
              }
            });
          }, 10);
          
        }, 100); // Increased delay for better stability
      } else {
        console.log('Table not found with ID: table-' + taskName);
      }
    }
